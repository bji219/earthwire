# LFO Nodes Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a standard LFO (Low Frequency Oscillator) node to Earthwire's channel processing pipeline, so any environmental data source can be paired with autonomous oscillation to add movement, rhythm, and expression.

**Architecture:** LFO is a new pipeline node inserted after the normalizer and before the smoother. It takes the normalized env-data value (0–1) and blends it with an oscillating waveform at a configurable depth — depth=0 is pure env data passthrough, depth=1 is pure LFO. Five shapes (sine, triangle, square, saw, rsaw) are supported. Rate is in Hz (free-running; BPM sync is a future addition). The node is factored identically to Smoother: a pure-TS factory in `src/lib/nodes/lfo.ts`, config interface in `types.ts`, wired in the engine, with UI controls in ChannelStrip.

**Tech Stack:** TypeScript, Svelte 4, Vitest (existing test runner)

// src/lib/nodes/lfo.ts
import type { LFOConfig, LFOShape } from './types.js';

export interface LFO {
  /** input: normalized env-data value 0–1. nowSec: current time in seconds (defaults to Date.now()/1000). */
  process(input: number, nowSec?: number): number;
  reset(): void;
}

function shapeValue(shape: LFOShape, phase: number): number {
  switch (shape) {
    case 'sine':     return Math.sin(phase * Math.PI * 2) * 0.5 + 0.5;
    case 'triangle': return phase < 0.5 ? phase * 2 : 2 - phase * 2;
    case 'square':   return phase < 0.5 ? 1 : 0;
    case 'saw':      return phase;
    case 'rsaw':     return 1 - phase;
  }
}

export function createLFO(config: LFOConfig): LFO {
  let startTime: number | null = null;

  function process(input: number, nowSec?: number): number {
    const t = nowSec ?? Date.now() / 1000;
    if (startTime === null) startTime = t;
    const phase = ((t - startTime) * config.rate) % 1;
    const osc = shapeValue(config.shape, phase);
    // Linear interpolation: depth=0 → pure input, depth=1 → pure LFO
    return input + (osc - input) * config.depth;
  }

  function reset(): void {
    startTime = null;
  }

  return { process, reset };
}
```

function toggleLFO() {
  updateChannel({
    lfo: channel.lfo ? null : { shape: 'sine', rate: 0.5, depth: 0.5 }
  });
}

function handleLFOShape(e: Event) {
  const shape = (e.target as HTMLSelectElement).value as LFOShape;
  updateChannel({ lfo: { ...channel.lfo!, shape } });
}

function handleLFORate(e: Event) {
  const rate = parseFloat((e.target as HTMLInputElement).value);
  updateChannel({ lfo: { ...channel.lfo!, rate } });
}

function handleLFODepth(e: Event) {
  const depth = parseFloat((e.target as HTMLInputElement).value);
  updateChannel({ lfo: { ...channel.lfo!, depth } });
}

$: lfoRateDisplay = channel.lfo
  ? channel.lfo.rate < 1
    ? `${(channel.lfo.rate * 1000).toFixed(0)}ms`
    : `${channel.lfo.rate.toFixed(2)}Hz`
  : '';
```
